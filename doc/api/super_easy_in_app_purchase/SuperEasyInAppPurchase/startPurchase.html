Future<void> startPurchase(String productID, {bool isConsumable = false}) async {
  for (var prod in _products) {
    if (_hasPurchased(prod.id) != null) {
      // Already purchased
      await _deliverProduct(prod.id);
    } else {
      _buyProduct(prod, isConsumable: isConsumable);
    }
  }
}
